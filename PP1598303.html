X = sm.add_constant(X)

# Dividir los datos en entrenamiento y prueba
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.3, random_state=42)

# Ajustar el modelo OLS (mtodo de mnimos cuadrados ordinarios) para eliminacin hacia atrs
def backward_elimination(X_train, y_train, significance_level=0.05):
    initial_model = sm.OLS(y_train, X_train).fit()
    while True:
        max_p_value = max(initial_model.pvalues)
        if max_p_value > significance_level:
            excluded_variable = initial_model.pvalues.idxmax()
            X_train = X_train.drop(columns=[excluded_variable])
            initial_model = sm.OLS(y_train, X_train).fit()
        else:
            break
    return initial_model, X_train

# Ejecutar eliminacin hacia atrs
model, X_selected = backward_elimination(X_train, y_train)

# Ver los resultados de eliminacin hacia atrs
print(f'Nmero de variables antes de eliminacin: {X.shape[1]}')
print(f'Nmero de variables despus de eliminacin: {X_selected.shape[1]}')
print("Variables seleccionadas:")
print(X_selected.columns)

# Aplicar Lasso para regularizacin
lasso = LassoCV(cv=5, random_state=42)

# Resultados de Lasso
print(f"Coeficientes de Lasso para las variables seleccionadas: {lasso.coef_}")
